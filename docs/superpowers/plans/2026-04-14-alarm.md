# Alarm Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a one-shot wall-clock alarm as a fourth timer kind, surfaced as a tab in `/tools/timer` (first in the pill bar) and as a sticky home widget with auto-opening popup on ring.

**Architecture:** Extend the existing `TimerContext` reducer/provider with an `AlarmTimer` kind that stores an absolute epoch `fireAt`. The 100ms ticker dispatches `alarm/tick` while armed; the reducer transitions to ringing when `Date.now() >= fireAt`. A module-level looping chime in `useTimer.ts` is started on `ringing: true` and stopped by `STOP`. UI mirrors the existing timer widgets/popup with a non-dismissable popup variant during ringing.

**Tech Stack:** React 18 + TypeScript, framer-motion, Radix UI Dialog, lucide-react, Web Audio API, vitest.

**Spec:** `docs/superpowers/specs/2026-04-14-alarm-design.md`

```ts
export const ALARM_DEFAULT_TIME = '12:30';

/**
 * Given an HH:MM (24h) target and a current epoch ms `now`, return the absolute
 * epoch ms when the alarm should fire. If the wall-clock time is in the past
 * (or equal to the current minute), schedule for tomorrow.
 */
export function computeFireAt(targetTime: string, now: number): number {
  const [h, m] = targetTime.split(':').map((s) => parseInt(s, 10));
  const target = new Date(now);
  target.setHours(h, m, 0, 0);
  if (target.getTime() <= now) target.setDate(target.getDate() + 1);
  return target.getTime();
}
```

```ts
const STORAGE_KEY = 'home-timers-v4';

// inside useReducer initializer:
const [timers, dispatch] = useReducer(reducer, undefined, () => {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return makeDefaults();
    const parsed = JSON.parse(raw) as TimerInstance[];
    const defaults = makeDefaults();
    // Merge: for each kind in defaults, prefer parsed if present.
    return defaults.map((d) => {
      const found = parsed.find((p) => p.kind === d.kind);
      if (!found) return d;
      const base = { ...found, running: false, startedAt: null };
      if (base.kind === 'pomodoro') {
        return {
          ...base,
          settings: {
            focusMin: clamp(base.settings?.focusMin ?? 25, 1, 180),
            pauseMin: clamp(base.settings?.pauseMin ?? 5, 1, 60),
            targetCycles: clamp(base.settings?.targetCycles ?? 4, 1, 12),
          },
        };
      }
      if (base.kind === 'alarm') {
        // Don't restore mid-ring or armed state across reloads — start unarmed.
        return { ...base, fireAt: null, ringing: false };
      }
      return base;
    });
  } catch {
    return makeDefaults();
  }
});
```

```ts
function applyAction(t: TimerInstance, a: Action): TimerInstance {
  if (a.type === 'load') return a.timers.find((nt) => nt.kind === t.kind) ?? t;
  if (a.type === 'setPersistent' && a.kind === t.kind) return { ...t, persistent: a.persistent };
  if (a.type === 'setColor' && a.kind === t.kind) return { ...t, color: a.color };

  if (t.kind === 'countdown') return reduceCountdown(t, a);
  if (t.kind === 'stopwatch') return reduceStopwatch(t, a);
  if (t.kind === 'pomodoro') return reducePomodoro(t, a);
  if (t.kind === 'alarm') return reduceAlarm(t, a);
  return t;
}
```

Add the `reduceAlarm` function after `reducePomodoro`:

```ts
function reduceAlarm(t: AlarmTimer, a: Action): AlarmTimer {
  switch (a.type) {
    case 'alarm/setTime':
      return { ...t, targetTime: a.time };
    case 'alarm/arm': {
      const fireAt = computeFireAt(t.targetTime, a.now);
      return {
        ...t,
        fireAt,
        running: true,
        startedAt: a.now,
        zeroedAt: null,
        ringing: false,
        lastSetTime: t.targetTime,
      };
    }
    case 'alarm/cancel':
      return { ...t, fireAt: null, running: false, startedAt: null, ringing: false };
    case 'alarm/tick': {
      if (!t.running || t.fireAt === null) return t;
      if (a.now < t.fireAt) return t;
      return {
        ...t,
        ringing: true,
        running: false,
        startedAt: null,
        fireAt: null,
        zeroedAt: a.now,
      };
    }
    case 'alarm/stop':
      return { ...t, ringing: false, running: false, startedAt: null, fireAt: null };
    default:
      return t;
  }
}
```

```ts
let alarmLoopId: ReturnType<typeof setInterval> | null = null;

/**
 * Start playing the countdown chime on a 1.5s loop. Idempotent — calling
 * twice doesn't stack intervals. Used by the alarm feature, which loops
 * until the user clicks STOP.
 */
export function startLoopingAlarm(): void {
  stopLoopingAlarm();
  playAlarm();
  alarmLoopId = setInterval(() => playAlarm(), 1500);
}

/** Stop the looping chime. Safe to call when not running. */
export function stopLoopingAlarm(): void {
  if (alarmLoopId !== null) {
    clearInterval(alarmLoopId);
    alarmLoopId = null;
  }
}
```
